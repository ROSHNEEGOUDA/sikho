    input_shape=(IMG_SIZE, IMG_SIZE, 3),
    include_top=False,

def plot_history(h1, h2=None):
    plt.figure(figsize=(12,5))

    # Accuracy
    plt.subplot(1,2,1)
    plt.plot(h1.history['accuracy'], label='train_acc')
    plt.plot(h1.history['val_accuracy'], label='val_acc')
    if h2:
        plt.plot(h2.history['accuracy'], label='fine_train_acc')
        plt.plot(h2.history['val_accuracy'], label='fine_val_acc')
    plt.title('Accuracy')
    plt.legend()

    # Loss
    plt.subplot(1,2,2)
    plt.plot(h1.history['loss'], label='train_loss')
    plt.plot(h1.history['val_loss'], label='val_loss')
    if h2:
        plt.plot(h2.history['loss'], label='fine_train_loss')
        plt.plot(h2.history['val_loss'], label='fine_val_loss')
    plt.title('Loss')
    plt.legend()

    plt.show()
